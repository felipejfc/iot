cmake_minimum_required(VERSION 3.20.0)

find_package(Zephyr REQUIRED HINTS $ENV{ZEPHYR_BASE})

project(light_switch-lp)

# NORDIC SDK APP START
target_sources(app PRIVATE
  src/main.c
  src/gpio_control.c
  src/zigbee_device.c
  src/zigbee_handlers.c
  src/adc_reader.c
)

# Use custom button handler only when DK library is not enabled
target_sources_ifndef(CONFIG_DK_LIBRARY app PRIVATE
  src/button_handler.c
)

target_include_directories(app PRIVATE include)
# NORDIC SDK APP END

target_sources_ifdef(CONFIG_BT_NUS app PRIVATE
  src/nus_cmd.c
)
